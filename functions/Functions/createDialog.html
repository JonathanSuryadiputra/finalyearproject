AlertDialog createDialog(String message, context) {
  // set up the ok button
  Widget okButton = TextButton(
    child: Text("OK"),
    onPressed: () {
      Navigator.pop(context);
    },
  );
  // set up the message dialog
  AlertDialog alert = AlertDialog(
    title: Text(message),
    actions: [
      okButton, // put the ok button as one of the action buttons
    ],
  );
  return alert;
}
